export const meta = {
	slug: 'laha',
	name: 'Laha',
	category: ['Beauty', 'Branding', 'Web'],
	description: 'Skincare clinic transformation.',
	sub: 'Branding',
	year: 2020,
	preview: '03',
	width: 'half',
	order: 3,
	gradientFrom: '#EAE1D6',
	gradientTo: '#C1C0E3',
};

export async function getStaticProps(context) {
	const images = await getAllImage('laha');
	const { gradientFrom, gradientTo } = meta;
	for (const key in images) {
		images[key] = { ...images[key], gradientFrom, gradientTo };
	}
	const ogImage =
		(process.env.HOST || '') + '/images/home/' + meta.preview + '.png';
	const title = rootMeta.name + ' | ' + meta.name;
	return { props: { images, ...meta, title, ogImage } };
}
